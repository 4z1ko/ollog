**Replace the entire `<script>` block (lines 115-159) with:**

```html
<script>
  const NOTIFY_SOUND = "{{ 'true' if notify_sound else 'false' }}";
  (function () {
    var indicator = document.getElementById('live-indicator');
    var eventsFlowing = false;
    var AudioCtxClass = window.AudioContext || window.webkitAudioContext;
    var audioCtx = null;
    var userInteracted = false;

    function onFirstInteraction() {
      if (userInteracted) return;
      userInteracted = true;
      if (AudioCtxClass) {
        audioCtx = new AudioCtxClass();
      }
    }

    document.addEventListener('click', onFirstInteraction);
    document.addEventListener('keydown', onFirstInteraction);

    async function playTone(ctx) {
      if (ctx.state === 'suspended') {
        await ctx.resume();
      }
      var now = ctx.currentTime;
      var osc = ctx.createOscillator();
      var gain = ctx.createGain();

      osc.type = 'sine';
      osc.frequency.setValueAtTime(440, now);

      gain.gain.setValueAtTime(0, now);
      gain.gain.linearRampToValueAtTime(0.3, now + 0.01);
      gain.gain.linearRampToValueAtTime(0.3, now + 0.08);
      gain.gain.linearRampToValueAtTime(0, now + 0.11);

      osc.connect(gain);
      gain.connect(ctx.destination);

      osc.start(now);
      osc.stop(now + 0.12);
    }

    // htmx:sseOpen detail has {source} but no .elt -- use e.target instead.
    // Do NOT set green on open; wait for the first actual new_qso event.

    document.body.addEventListener('htmx:sseMessage', function (e) {
      // e.detail is the raw MessageEvent; e.target is the element that has
      // hx-trigger="sse:new_qso" (i.e. #log-table).
      if (!e.target || e.target.id !== 'log-table') return;
      if (!e.detail || e.detail.type !== 'new_qso') return;
      if (!eventsFlowing) {
        eventsFlowing = true;
        indicator.classList.remove('hidden');
        indicator.classList.add('flex');
        indicator.querySelector('span:last-child').textContent = 'LIVE';
        indicator.classList.remove('bg-rose-100', 'text-rose-700');
        indicator.classList.add('bg-emerald-100', 'dark:bg-emerald-900/40',
                                'text-emerald-700', 'dark:text-emerald-400');
      }
      // Sound notification -- BEFORE auto-refresh guards so tone fires
      // independently of page/filter state (SND-01 + RESEARCH.md Open Question)
      if (NOTIFY_SOUND === 'true' && userInteracted && audioCtx) {
        playTone(audioCtx);
      }
      if (!document.getElementById('auto-refresh-ok')) return;
      if (document.querySelector('#log-table input')) return;
      htmx.ajax('GET', '/log/view', { target: '#log-table', swap: 'innerHTML' });
    });

    document.body.addEventListener('htmx:sseError', function (e) {
      if (e.target && e.target.id === 'log-table') {
        eventsFlowing = false;
        indicator.classList.remove('hidden');
        indicator.classList.add('flex');
        indicator.querySelector('span:last-child').textContent = 'OFFLINE';
      }
    });

    document.body.addEventListener('htmx:sseClose', function (e) {
      if (e.target && e.target.id === 'log-table') {
        eventsFlowing = false;
        indicator.classList.add('hidden');
        indicator.classList.remove('flex');
      }
    });
  })();
</script>
```

## Trust Boundaries

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-46-01 | Tampering | `NOTIFY_SOUND` JS constant | accept | Value is a string `"true"/"false"` rendered server-side from `user.notify_sound` (a Python bool from Beanie). No user input reaches the Jinja2 injection point. An attacker who modifies the constant in DevTools can only toggle their own sound -- no security impact. |
| T-46-02 | Information Disclosure | `notify_sound` in template context | accept | The preference is a non-sensitive boolean (sound on/off). It is rendered only to the authenticated operator's own page via `get_current_user_cookie`. No cross-operator leakage possible. |

**No new attack surface introduced.** The dependency swap from `get_current_operator_callsign_cookie` to `get_current_user_cookie` uses the same JWT HttpOnly cookie validation path already used by `profile_page()`, `submit_qso()`, `tokens_list()`, `tokens_create()`, and `tokens_revoke()` in the same file. The `callsign = user.callsign` extraction preserves identical operator isolation. No new endpoints, no new data ingestion, no new auth paths.
