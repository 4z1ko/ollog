<script>
  var PALETTES = {
    dark:  ['#818cf8','#34d399','#fbbf24','#60a5fa','#f472b6','#a78bfa','#2dd4bf','#fb923c'],
    light: ['#4f46e5','#059669','#d97706','#2563eb','#db2777','#7c3aed','#0d9488','#ea580c']
  };

  var bandData   = {{ band_counts   | tojson }};
  var modeData   = {{ mode_counts   | tojson }};
  var entityData = {{ entity_counts | tojson }};

  function makeChart(canvasId, labels, values, palette, textColor) {
    var canvas = document.getElementById(canvasId);
    if (!canvas) return;
    var existing = Chart.getChart(canvas);
    if (existing) existing.destroy();
    return new Chart(canvas, {
      type: 'pie',
      data: {
        labels: labels,
        datasets: [{
          data: values,
          backgroundColor: values.map(function(_, i) { return palette[i % palette.length]; }),
          borderWidth: 2,
          borderColor: 'transparent'
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          legend: {
            display: true,
            position: 'bottom',
            labels: { color: textColor, padding: 16 }
          }
        }
      }
    });
  }

  function initCharts() {
    var isDark = document.documentElement.classList.contains('dark');
    var palette = isDark ? PALETTES.dark : PALETTES.light;
    var textColor = isDark ? '#9ca3af' : '#6b7280';

    makeChart('chart-band',
      Object.keys(bandData), Object.values(bandData), palette, textColor);
    makeChart('chart-mode',
      Object.keys(modeData), Object.values(modeData), palette, textColor);
    makeChart('chart-entity',
      entityData.map(function(e) { return e.name; }),
      entityData.map(function(e) { return e.count; }),
      palette, textColor);
  }

  function reinitCharts() { initCharts(); }

  document.addEventListener('DOMContentLoaded', initCharts);
  window.addEventListener('themechange', reinitCharts);
</script>
{% endblock %}
{% endif %}

Key design decisions implemented:
- **D-03 (tojson safety):** All three data variables use `| tojson` filter -- never `| safe` or bare substitution. Entity names contain commas, quotes, apostrophes.
- **D-04 (canvas sizing):** Each `<canvas>` is wrapped in `<div class="relative h-64 w-full">` with `maintainAspectRatio: false` in Chart.js options.
- **D-05 (stale canvas guard):** `Chart.getChart(canvas)?.destroy()` called before every `new Chart()` (using `var existing = Chart.getChart(canvas); if (existing) existing.destroy();` for broad browser compatibility).
- **D-08 (page width):** `max-w-5xl mx-auto`.
- **D-09 (summary placement):** "Total QSOs: N" in summary card above charts. "N entities" inline in DXCC chart title as `&middot; {{ unique_entity_count }} entities`.
- **D-10 (chart grid layout):** Band + Mode in `grid grid-cols-1 md:grid-cols-2 gap-6` top row; DXCC Entity full-width below.
- **Chart.js CDN (D-02):** UMD bundle from `https://cdn.jsdelivr.net/npm/chart.js@4.5.1/dist/chart.umd.min.js` loaded only in extra_scripts block.
- **Empty state:** `{% if total_qsos == 0 %}` renders card with empty-state message and no canvas elements; Chart.js script tag is conditionally guarded by `{% if total_qsos > 0 %}`.
- **Dark mode palettes (from UI-SPEC):** `PALETTES.dark` = `['#818cf8','#34d399','#fbbf24','#60a5fa','#f472b6','#a78bfa','#2dd4bf','#fb923c']`, `PALETTES.light` = `['#4f46e5','#059669','#d97706','#2563eb','#db2777','#7c3aed','#0d9488','#ea580c']`.
- **Chart text colors:** Dark mode `'#9ca3af'` (gray-400), light mode `'#6b7280'` (gray-500).
- **Color cycling for >8 slices:** `values.map(function(_, i) { return palette[i % palette.length]; })` prevents missing colors.
- **Theme listener (D-06):** `window.addEventListener('themechange', reinitCharts)` re-creates all charts with updated palette on theme toggle.
- **Accessibility:** Each canvas has `role="img"` and descriptive `aria-label`. Chart.js legend is enabled at `position: 'bottom'`.
- **Typography (from UI-SPEC):** Page heading `text-xl font-semibold` (heading weight 600 per UI-SPEC typography table).
