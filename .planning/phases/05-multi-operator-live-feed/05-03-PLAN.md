**2. Create `app/feed/manager.py`** — ConnectionManager class:

```python
import asyncio

class ConnectionManager:
    def __init__(self):
        self._queues: set[asyncio.Queue] = set()

    async def connect(self) -> asyncio.Queue:
        q: asyncio.Queue = asyncio.Queue()
        self._queues.add(q)
        return q

    def disconnect(self, q: asyncio.Queue) -> None:
        self._queues.discard(q)

    async def broadcast(self, event: str) -> None:
        for q in set(self._queues):  # snapshot to avoid mutation during iteration
            await q.put(event)

manager = ConnectionManager()
```

Note: `broadcast()` accepts a `str` (rendered HTML), not a dict. The change stream watcher renders HTML before broadcasting.

Add a `watch_qsos` async function in the same file:

```python
import asyncio
import logging
from pymongo.errors import PyMongoError

logger = logging.getLogger(__name__)

async def watch_qsos(collection, mgr: ConnectionManager, templates) -> None:
    """Watch the qsos collection for inserts and broadcast rendered HTML to all SSE clients."""
    pipeline = [{"$match": {"operationType": "insert"}}]
    while True:
        try:
            async with await collection.watch(pipeline, full_document="updateLookup") as stream:
                async for change in stream:
                    doc = change.get("fullDocument", {})
                    if doc:
                        ctx = {
                            "call": doc.get("CALL", ""),
                            "band": doc.get("BAND", ""),
                            "mode": doc.get("MODE", ""),
                            "freq": doc.get("FREQ", ""),
                            "operator": doc.get("_operator", ""),
                            "qso_date_utc": doc.get("qso_date_utc"),
                        }
                        # Render HTML partial — no Request object needed
                        html = templates.get_template("log/feed_row.html").render(ctx)
                        await mgr.broadcast(html)
        except PyMongoError as e:
            logger.warning("Change stream error, reconnecting: %s", e)
            await asyncio.sleep(1)
        except asyncio.CancelledError:
            logger.info("Change stream watcher cancelled")
            break
```

Key details:
- Use `full_document="updateLookup"` in `watch()` to ensure `fullDocument` is populated for inserts (it is by default for inserts, but explicit is clearer).
- The `CancelledError` handler breaks the reconnect loop cleanly.
- Use `templates.get_template(...).render(ctx)` (NOT `TemplateResponse`) — no Request object in scope.

**3. Create `app/feed/router.py`** — SSE endpoint:

```python
from collections.abc import AsyncIterable
from fastapi import APIRouter, Depends
from fastapi.sse import EventSourceResponse, ServerSentEvent
from app.auth.dependencies import get_current_operator_callsign_cookie
from app.feed.manager import manager

router = APIRouter(prefix="/feed", tags=["feed"])

@router.get("/station", response_class=EventSourceResponse)
async def station_feed(
    _callsign: str = Depends(get_current_operator_callsign_cookie),
) -> AsyncIterable[ServerSentEvent]:
    q = await manager.connect()
    try:
        while True:
            html = await q.get()
            yield ServerSentEvent(data=html, event="new_qso")
    finally:
        manager.disconnect(q)
```

Key details:
- `_callsign` is used for auth gating only (underscore prefix = unused).
- Cookie auth via `get_current_operator_callsign_cookie` — EventSource API cannot send Bearer headers.
- Event name is `"new_qso"` matching the `sse-swap="new_qso"` in the template.
- `finally` block ensures queue cleanup on client disconnect.

e) Update the lifespan function to start and cancel the watcher task:
```python
@asynccontextmanager
async def lifespan(app: FastAPI):
    await init_db()
    await _bootstrap_admin()
    # Start change stream watcher for live feed
    client = get_client()
    watcher_task = None
    if client is not None:
        collection = client[settings.mongodb_db]["qsos"]
        watcher_task = asyncio.create_task(
            watch_qsos(collection, feed_manager, _templates)
        )
    yield
    if watcher_task is not None:
        watcher_task.cancel()
        try:
            await watcher_task
        except asyncio.CancelledError:
            pass
    await close_db()
```

Key details:
- The watcher task is started AFTER `init_db()` and `_bootstrap_admin()`.
- The task is cancelled BEFORE `close_db()` to prevent the watcher from using a closed connection.
- If `get_client()` returns None (shouldn't happen after init_db, but defensive), skip the watcher.
