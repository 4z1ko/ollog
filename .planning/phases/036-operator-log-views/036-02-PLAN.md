```html
          <tbody>

              <td>{{ item.record_index + 1 }}</td>
              <td>{{ item.call }}</td>
            </tr>
            {% endfor %}
          </tbody>
        </table>
      </div>
    </div>
    {% endif %}

    {% if report.duplicates %}
    <div>
      <h3 class="text-xs font-semibold uppercase tracking-wider text-amber-700 dark:text-amber-400 mt-3 mb-1.5">
        Duplicates ({{ report.duplicates | length }})
      </h3>
      <div class="table-wrap">
        <table class="data-table">
          <thead>
            <tr>
              <th>Record #</th>
              <th>Callsign</th>
              <th>Existing ID</th>
            </tr>
          </thead>
          <tbody>
            {% for item in report.duplicates %}
            <tr>
              <td>{{ item.record_index + 1 }}</td>
              <td>{{ item.call }}</td>
              <td>{{ item.existing_id }}</td>
            </tr>
            {% endfor %}
          </tbody>
        </table>
      </div>
    </div>
    {% endif %}

    {% if report.errors %}
    <div>
      <h3 class="text-xs font-semibold uppercase tracking-wider text-rose-700 dark:text-rose-400 mt-3 mb-1.5">
        Errors ({{ report.errors | length }})
      </h3>
      <div class="table-wrap">
        <table class="data-table">
          <thead>
            <tr>
              <th>Record #</th>
              <th>Callsign</th>
              <th>Error</th>
            </tr>
          </thead>
          <tbody>
            {% for item in report.errors %}
            <tr>
              <td>{{ item.record_index + 1 if item.record_index is defined else "—" }}</td>
              <td>{{ item.call if item.call is defined else "—" }}</td>
              <td>{{ item.error }}</td>
            </tr>
            {% endfor %}
          </tbody>
        </table>
      </div>
    </div>
    {% endif %}

    {% if not report.accepted and not report.duplicates and not report.errors %}
    <p class="text-sm text-gray-500 dark:text-gray-400">No records found in file.</p>
    {% endif %}

  </div>
</div>
```

**Key decisions (do not deviate):**
- Outer wrapper: `.card` (not a bare div) — `.card` already carries dark-mode background, border, and shadow from input.css
- Inner wrapper: `.card-body space-y-4` — provides padding and vertical spacing between sections
- Summary line: `text-sm font-semibold text-gray-700 dark:text-gray-300` — no new component needed
- Section headings: raw Tailwind utilities, NOT badge classes (`.badge-amber` does not exist in the component library)
  - Accepted: `text-emerald-700 dark:text-emerald-400`
  - Duplicates: `text-amber-700 dark:text-amber-400`
  - Errors: `text-rose-700 dark:text-rose-400`
- Tables: each inside `<div class="table-wrap">` wrapping `<table class="data-table">` — `.table-wrap` is required for rounded border/overflow behavior
- Empty fallback: `text-sm text-gray-500 dark:text-gray-400`
- No new classes added to input.css — all classes above are standard Tailwind v3 or existing Phase 33 components
