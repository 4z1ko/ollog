import bisect

def _range_lookup(prefix: str) -> str | None:
    idx = bisect.bisect_right(_STARTS, prefix) - 1
    if idx < 0:

    start, end, iso = _RANGES[idx]
    if start <= prefix <= end:
        return iso
    return None
